noop是空函数 开发时引入Flow类，然后按以下步骤创建活动 可参考sohu_test和qunar5_test两个项目


  import Flow from 'commonJsDir/Flow';

  var flowWorker = Flow()
  //使用use方法动态给flowWorker对象添加函数，调用该函数会延迟到模板参数（getTmpl接口）请求返回之后才能执行
  .use('xx',noop)
  //set动态添加的方法在任何时候都能马上执行
  //返回抽奖奖品ID
  .set('getAwardId',noop)
  //设置下载进度文案
  .set('setProgressTxt', function($scope,value){
    $('#dl-prg-txt').text(value);
  }, 'downloadManage')
  //只会在客户端的下载回调函数中执行一次
  .set('startDownload',noop,'downloadManage')
   //每次进入客户端的下载回调函数时都会执行
  .set('downloadNotify',function($scope,evtObj){
    var btnTxt = evtObj.btnText,
        btnStatus = evtObj.btnStatus;

    switch(btnStatus){
      case 'neverInstallStatus':
        $('#dl-prg-txt').text(Constant.appSize);
        break;
    }
  },'downloadManage')
    .init(noop)
    //业务逻辑都以状态机方式运行
    .business(stateManager);
